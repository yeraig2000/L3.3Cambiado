	ListaSockets sockets;

//Añade un socket a la Lista de Sockets
int Pon (ListaSockets *lista, int socket){
		if(lista->num==100)
			return -1;
		else {
				lista->sockets[lista->num].socket = socket;
		lista->num++;
		return 0;
		}
	}
//Devuelve la posición que ocupa un socket determinado
int DamePosicion(ListaSockets *lista, int socket){
int i=0;
int encontrado = 0;
while ((i<lista->num) && !encontrado)
		{
			if(lista->sockets[i].socket == socket)
			encontrado =1;
if (!encontrado)
	i=i+1;
	}
	if (encontrado)
		return i;
	else
		return -1;
}

//Elimina un socket que ya no se está utilizando
int Eliminar (ListaSockets *lista, int socket)
{
		int pos = DamePosicion (lista,socket);
			if (pos == -1)
				return -1;
			else{
					int i;
					for (i=pos; i<lista->num-1; i++)
					{
						lista->sockets[i] = lista->sockets[i+1];
						}
						lista->num--;
						return 0;
						}
	}
